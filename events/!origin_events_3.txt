namespace = origin

#Lithoid origin
planet_event = {
	id = origin.3010
	hide_window = yes

	is_triggered_only = yes
	
	trigger = {
		FROM = { 
			exists = FROM
			is_ship_size = lithoid_colonizer
		} 
		NOR = { # Prevent lithoid meteor spamming
			has_deposit = d_lithoid_crater_small
			has_deposit = d_lithoid_crater_large 
			has_modifier = lithoid_crater2
		}
	}

	immediate = {
		add_deposit = d_buried_lithoids
		add_deposit = d_buried_lithoids
		if = {
			limit = {
				NOR = {
					is_planet_class = pc_habitat
					is_planet_class = pc_ringworld_habitable
				}
			}
			add_deposit = d_lithoid_crater_small
			add_modifier = { modifier = lithoid_crater2 }
		}
	}
}
#this makes it not infinitly create shit
planet_event = {
	id = origin.3003
	title = "origin.3003.name"
	desc = "origin.3003.desc"
	picture = GFX_evt_burning_city
	show_sound = event_alien_nature

	is_triggered_only = yes

	trigger = {
		planet_devastation >= 50
		OR = {
			has_deposit = d_tree_of_life_colony
			has_deposit = d_tree_of_life_home
		}
		NOT = { has_modifier = no_lifetree }
	}

	immediate = {
		remove_deposit = d_tree_of_life_colony
		remove_deposit = d_tree_of_life_home
	}

	option = {
		name = UNFORTUNATE
		tooltip = {
			remove_deposit = d_tree_of_life_colony
			remove_deposit = d_tree_of_life_home
		}
		add_modifier = { modifier = no_lifetree }
	}
}
